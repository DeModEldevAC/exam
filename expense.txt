import 'package:flutter/material.dart';
import 'package:get/get.dart';
import 'moneyController.dart';

class Addpage extends StatelessWidget {
  Addpage({super.key});

  final MoneyController controller = Get.find();

  String type = "Expense"; // default

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text("Add Transaction")),

      body: Padding(
        padding: EdgeInsets.all(20),
        child: Column(
          children: [
            // Amount input
            TextField(
              controller: controller.amountController,
              keyboardType: TextInputType.number,
              decoration: InputDecoration(
                hintText: "Enter Amount",
                border: OutlineInputBorder(),
              ),
            ),

            SizedBox(height: 20),

            // Dropdown
            DropdownButton<String>(
              value: type,
              items: [
                DropdownMenuItem(value: "Income", child: Text("Income")),
                DropdownMenuItem(value: "Expense", child: Text("Expense")),
              ],
              onChanged: (value) {
                type = value!;
              },
            ),

            SizedBox(height: 20),

            ElevatedButton(
              onPressed: () {
                int? a = int.tryParse(controller.amountController.text);
                int amount = a != null ? a : 0;
                controller.addTransaction(amount, type);

                Get.back();
              },
              child: Text("Add"),
            ),
          ],
        ),
      ),
    );
  }
}


















import 'package:flutter/material.dart';
import 'package:flutter_application_1/transactionModel.dart';
import 'package:get/get.dart';

class MoneyController extends GetxController {
  RxInt availableBalance = 0.obs;
  RxInt incomeBalance = 0.obs;
  RxInt expenseBalance = 0.obs;
  TextEditingController amountController = TextEditingController();
  RxList<TransactionModel> transactions = <TransactionModel>[].obs;

  void addTransaction(int amount, String type) {
    transactions.add(TransactionModel(title: type, amount: amount, type: type));

    if (type == "Expense") {
      expenseBalance.value += amount;
      availableBalance.value -= amount;
    } else {
      incomeBalance.value += amount;
      availableBalance.value += amount;
    }
  }
}














class TransactionModel {
  String title;
  int amount;
  String type; // "Income" or "Expense"

  TransactionModel({
    required this.title,
    required this.amount,
    required this.type,
  });
}
